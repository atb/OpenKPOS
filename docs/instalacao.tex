\chapter{Instalação e Configuração}
\label{ch:instalacao}

\indent A instalação e configuração deve ser sempre efectuada por técnicos da
KTC.
\newline
\newline
\noindent A instalação contempla:
\begin{itemize}
	\item hardware (periférios)
	\item java
	\item servidor de base de dados
\end{itemize}
\noindent A configuração contempla:
\begin{itemize}
	\item configuração dos diversos postos
	\item dados de empresa
	\item armazéns
	\item \emph{layout} dos documentos
	\item utilizadores e permissões
\end{itemize}
